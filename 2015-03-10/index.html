handleRequest(HttpRequest request){
  if (request.uri.path == '/') {
    request.response..write("Welcome to my todos")
                    ..close();
  } else if (request.uri.path == "/api/todos") {
    request.response..write(JSON.encode(CONTACTS))
                    ..close();
  } else {
    request.response..statusCode = HttpStatus.NOT_FOUND
                    ..close();
  }
}

lang-dartimport 'package:shelf/shelf.dart';
import 'package:shelf_appengine/shelf_appengine.dart';
import 'package:shelf_route/shelf_route.dart';

final itemsUrl = new UriParser(new UriTemplate('/api/todos'));
final oneItemUrl = new UriParser(new UriTemplate('/api/todos/{todoId}'));
final pingUrl = new UriParser(new UriTemplate('/api/ping'));

void main() {
  var apiRouter = router()
        ..post(itemsUrl, createTodo)
        ..get(ping, (request) => new Response.ok("PONG"))
        ..delete(oneItemUrl, (request) { 
          deleteItem(int.parse(getPathParameter(request, todoId));
          return new Response(204);
        });
   
  var cascade = new Cascade()
    .add(apiRouter.handler)
    .add(assetHandler(directoryIndexServeMode: DirectoryIndexServeMode.REDIRECT));
  
  serve(cascade.handler);
}

Cloud storage

Concepts

  • Service to store data in Google's cloud.
  • High Capacity and Scalability.
  • Accessible from all instances.
    • C'est le service de stockage de données de Google
    • Grande capacité (<5To docment) et hautement scalable
    • données accessibles depuis instances (backup, script, ...)

Terminology

  • Buckets are the basic containers that hold your data.
  • Objects are the individual pieces of data that you store.
